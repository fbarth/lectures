\begin{Slide}{Definição de dados}
"Data are values of qualitative or quantitative variables, belonging
to a set of items.''

http://en.wikipedia.org/wiki/Data

\newpage
"Data are values of qualitative or quantitative variables, belonging
to a \emph{set of items}.''

\emph{Set of items}: conjunto de itens (objetos) de interesse.

\newpage

"Data are values of qualitative or quantitative \emph{variables}, belonging
to a set of items.''

\emph{variables}: uma medida ou uma característica de um item.

\newpage

"Data are values of \emph{qualitative} or \emph{quantitative} variables, belonging
to a set of items.''

\emph{qualitative}: cidade de origem, sexo, fez ou não tratamento.

\emph{quantitative}: peso, altura, pressão do sangue.

\end{Slide}

\begin{Slide}{Raw data versus dados processados}
\textbf{Raw data}
\begin{itemize}
\item Fonte original dos dados
\item Geralmente difícil para fazer algum tipo de análise
\end{itemize}

http://en.wikipedia.org/wiki/Raw\_Data

\newpage

\textbf{Dados processados}
\begin{itemize}
\item Dados que estão prontos para serem analisados
\item O processamento pode incluir \textit{merging},
  \textit{subsetting}, \textit{transforming}, etc...
\item Todas as etapas devem ser registradas
\end{itemize}

http://en.wikipedia.org/wiki/Compute\_data\_processing
\end{Slide}

\begin{Slide}{Missing values}

\small
\begin{alltt}

vector1

## [1] 188.2 181.3 193.4 NA

is.na(vector1)

## [1] FALSE FALSE FALSE TRUE
\end{alltt}
\end{Slide}

\begin{Slide}{Subsetting}
